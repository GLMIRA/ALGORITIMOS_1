    real altura_tanque 

    real raio_tanque 

    real area_tanque 

    real litros_por_lata = 5 

    real rendimento_por_litro = 3   

    real preco_lata 

    real quantidade_latas 

    real custo_total 

    real PI = 3.14 

    real litros_necessarios 

    escreva("digite a altura do tanque em metros") 

    leia(altura_tanque) 

    escreva("digite o raio do tanque em metros") 

    leia(raio_tanque) 

    escreva("digite o preço da lata de tinta em reais") 

    leia(preco_lata) 

    area_tanque = (2 * PI * raio_tanque * (altura_tanque + raio_tanque)) 

    litros_necessarios = area_tanque / rendimento_por_litro 

    quantidade_latas = litros_necessarios / litros_por_lata 

    custo_total = preco_lata * quantidade_latas 
           
    escreva("a quantidade de latas de tinta necessária eh", quantidade_latas) 

    escreva("o custo total da pintura será: R$", custo_total) 
fim

    real ponto_1_eixo_y
    real potencia_ponto_1 

    real ponto_2_eixo_x
    real ponto_2_eixo_y
    real potencia_ponto_2


    escreva("digite o ponto 1 no eixo x")
    leia(ponto_1_eixo_x)
    escreva("digite o ponto 1 no eixo y")
    leia(ponto_1_eixo_y)

    escreva("digite o ponto 2 no eixo x")
    leia(ponto_2_eixo_x)
    escreva("digite o ponto 2 no eixo y")
    leia(ponto_2_eixo_y)

    potencia_ponto_1 = (ponto_1_eixo_x * ponto_1_eixo_y) * (ponto_1_eixo_x * ponto_1_eixo_y)
    potencia_ponto_2 = (ponto_2_eixo_x * ponto_2_eixo_y) * (ponto_2_eixo_x * ponto_2_eixo_y)

    distancia = sqrt(potencia_ponto_1 + potencia_ponto_2)

    escreva("a distancia eh de: ",distancia)
fim
